        const CONFIG = {
            storageKeyChat: 'gemini_chat_history',
            storageKeyConfig: 'gemini_app_config'
        };

        let state = {
            chatHistory: [],
            apiKey: '',
            model: 'gemini-2.5-flash',
            aiName: 'Gemini AI',
            theme: 'dark',
            isProcessing: false
        };

        // Cache UI elements

        /**
         * CORE APP STORAGE MANAGEMENT & INITIALIZATION
         */
        function initApp() {
            loadConfig();
            loadChatHistory();
            setupEventListeners();
            monitorNetworkStatus();
            renderChatWorkspace();
            applyTheme();
            detectSpeechCapability();
        }

        function loadConfig() {
            const rawConfig = localStorage.getItem(CONFIG.storageKeyConfig);
            if (rawConfig) {
                try {
                    const parsed = JSON.parse(rawConfig);
                    state.apiKey = parsed.apiKey || '';
                    state.model = parsed.model || 'gemini-2.5-flash';
                    state.aiName = parsed.aiName || 'Gemini AI';
                    state.theme = parsed.theme || 'dark';
                } catch(e) {
                    console.error("Failed to parse configurations, formatting fallback setup.", e);
                }
            }
            
            // Sync values to configurations input fields
            elements.apiKeyInput.value = state.apiKey;
            elements.modelSelect.value = state.model;
            elements.aiNameInput.value = state.aiName;
            elements.displayAiName.textContent = state.aiName;
        }

        function saveConfigState() {
            const systemConfig = {
                apiKey: state.apiKey,
                model: state.model,
                aiName: state.aiName,
                theme: state.theme
            };
            localStorage.setItem(CONFIG.storageKeyConfig, JSON.stringify(systemConfig));
            elements.displayAiName.textContent = state.aiName;
        }

        function loadChatHistory() {
            const rawChat = localStorage.getItem(CONFIG.storageKeyChat);
            if (rawChat) {
                try {
                    state.chatHistory = JSON.parse(rawChat);
                } catch(e) {
                    console.error("Failed parsing conversational data storage arrays.", e);
                    state.chatHistory = [];
                }
            }
        }

        function saveChatHistory() {
            localStorage.setItem(CONFIG.storageKeyChat, JSON.stringify(state.chatHistory));
        }

        function applyTheme() {
            elements.body.setAttribute('data-theme', state.theme);
        }

        function monitorNetworkStatus() {
            const updateStatus = () => {
                if (navigator.onLine) {
                    elements.statusIndicator.classList.remove('offline');
                    elements.networkWarning.style.display = 'none';
                } else {
                    elements.statusIndicator.classList.add('offline');
                    elements.networkWarning.style.display = 'block';
                }
            };
            window.addEventListener('online', updateStatus);
            window.addEventListener('offline', updateStatus);
            updateStatus(); // Initial run state execution
        }

        function detectSpeechCapability() {
            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            if (!SpeechRecognition) {
                elements.voiceBtn.style.display = 'none';
            }
        }

        /**
         * RENDER FLOW ARCHITECTURES
         */
        function renderChatWorkspace() {
            if (state.chatHistory.length === 0) {
                renderOnboardingState();
                return;
            }

            elements.chatWorkspace.innerHTML = '';
            state.chatHistory.forEach((msg) => {
                appendMessageBubbleUI(msg);
            });
            scrollToLatestMessage();
        }

        function appendMessageBubbleUI(msgObj) {
            // Remove onboarding dashboard frame if first message
            const onboarding = document.getElementById('onboarding-container');
            if (onboarding) onboarding.remove();

            const row = document.createElement('div');
            row.className = `message-row ${msgObj.role}`;
            row.id = `msg-${msgObj.id}`;

            const isAI = msgObj.role === 'ai';
            const actionButtonsHTML = isAI 
                ? `<button class="action-link" onclick="copyMessageText('${msgObj.id}')">Copy</button> | 
                   <button class="action-link" onclick="speakMessageText('${msgObj.id}')">Speak</button>`
                : '';

            row.innerHTML = `
                <div class="message-content">
                    <div class="message-bubble ${msgObj.isError ? 'system-error' : ''}">${escapeHTML(msgObj.text)}</div>
                    <div class="message-meta">
                        <span>${msgObj.timestamp}</span>
                        ${actionButtonsHTML}
                    </div>
                </div>
            `;

            elements.chatWorkspace.appendChild(row);
            scrollToLatestMessage();
        }

        /**
         * PROCESSING CONVERSATION FRAMEWORKS & API CONNECTIONS
         */
        async function submitUserMessage() {
            if (state.isProcessing) return;
            
            const rawText = elements.chatInput.value.trim();
            if (!rawText) return;

            // Log item structure creation
            const timestamp = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
            const messageId = 'msg-' + Date.now();
            
            const userMessage = {
                id: messageId,
                role: 'user',
                text: rawText,
                timestamp: timestamp,
                isError: false
            };

            // State mutations tracking render profiles updates
            state.chatHistory.push(userMessage);
            saveChatHistory();
            appendMessageBubbleUI(userMessage);

            // Clean input layout profiles
            elements.chatInput.value = '';
            elements.chatInput.style.height = 'auto';
            elements.charCounter.textContent = '0/2000';

            // Establish submission processing constraints
            setSubmissionLockState(true);
            renderTypingIndicator();

            try {
                const responseText = await queryGeminiAPIEngine(rawText);
                
                const aiMessage = {
                    id: 'msg-' + Date.now(),
                    role: 'ai',
                    text: responseText,
                    timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
                    isError: false
                };
                
                state.chatHistory.push(aiMessage);
                saveChatHistory();
                removeTypingIndicator();
                appendMessageBubbleUI(aiMessage);

            } catch (err) {
                removeTypingIndicator();
                const systemErrorMessage = {
                    id: 'msg-' + Date.now(),
                    role: 'ai',
                    text: `Error: ${err.message || 'An unexpected validation fault has stopped processing.'}`,
                    timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
                    isError: true
                };
                state.chatHistory.push(systemErrorMessage);
                appendMessageBubbleUI(systemErrorMessage);
            } finally {
                setSubmissionLockState(false);
            }
        }

        async function queryGeminiAPIEngine(prompt) {
            if (!state.apiKey) {
                throw new Error("Missing Gemini API Token Key configuration profile. Open the settings dashboard menu panel right above to assign authorization credentials before processing queries.");
            }

            // Using modern Gemini API operational endpoint routing parameters mapping structure
            const targetUrl = `https://generativelanguage.googleapis.com/v1beta/models/${state.model}:generateContent?key=${state.apiKey}`;
            
            const headers = {
                'Content-Type': 'application/json'
            };

            // Build structural context model matching historical multi-turn sequences standard specs format array profiles
            const bodyPayload = {
                contents: [
                    {
                        parts: [
                            { text: prompt }
                        ]
                    }
                ],
                generationConfig: {
                    temperature: 0.7,
                    maxOutputTokens: 2048
                }
            };

            const response = await fetch(targetUrl, {
                method: 'POST',
                headers: headers,
                body: JSON.stringify(bodyPayload)
            });

            if (!response.ok) {
                const dataError = await response.json().catch(() => ({}));
                const detailedErrorMsg = dataError.error?.message || `HTTP Execution error validation failure status response context: ${response.status}`;
                throw new Error(detailedErrorMsg);
            }

            const data = await response.json();
            
            // Structured extraction node traversal check matrix parsing arrays mapping validation loops
            if (data.candidates && data.candidates[0]?.content?.parts?.[0]?.text) {
                return data.candidates[0].content.parts[0].text;
            } else {
                throw new Error("Malformed JSON formatting properties object output detected on retrieval layer.");
            }
        }

        /**
         * INTERGATED BROWSER SPECIFIC UTILITIES API RUNTIMES
         */
        function copyMessageText(id) {
            const dataNode = state.chatHistory.find(x => x.id === id);
            if (!dataNode) return;
            
            navigator.clipboard.writeText(dataNode.text).then(() => {
                alert("Message text data systematically saved to runtime clip profile buffer clipboard memory arrays successfully.");
            }).catch(err => {
                console.error("Failing processing programmatic storage clipboard operations context parsing data node arrays: ", err);
            });
        }

        function speakMessageText(id) {
            const dataNode = state.chatHistory.find(x => x.id === id);
            if (!dataNode) return;

            // Terminate existing spoken utterances queues buffers processes cleanly first 
            window.speechSynthesis.cancel();

            const textToSpeechOutputNode = new SpeechSynthesisUtterance(dataNode.text);
            window.speechSynthesis.speak(textToSpeechOutputNode);
        }

        function executeVoiceCapturePipeline() {
            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            if (!SpeechRecognition) return;

            const audioAgentRecognizerEngine = new SpeechRecognition();
            audioAgentRecognizerEngine.lang = 'en-US';
            audioAgentRecognizerEngine.interimResults = false;
            audioAgentRecognizerEngine.maxAlternatives = 1;

            elements.voiceBtn.style.color = '#ea4335'; // Visual record cue indication marker state

            audioAgentRecognizerEngine.onresult = (event) => {
                const vocalParseOutputText = event.results[0][0].transcript;
                elements.chatInput.value = (elements.chatInput.value + " " + vocalParseOutputText).trim();
                elements.chatInput.dispatchEvent(new Event('input'));
            };

            audioAgentRecognizerEngine.onerror = (e) => {
                console.error("Microphone Capture validation tracking metrics failure routing error context details:", e.error);
                alert("Microphonic processing interface engine run capture failure configuration runtime interrupt code verified error sequence tracking: " + e.error);
            };

            audioAgentRecognizerEngine.onend = () => {
                elements.voiceBtn.style.color = 'inherit';
            };

            audioAgentRecognizerEngine.start();
        }

        function escapeHTML(str) {
            return str.replace(/[&<>'"]/g, 
                tag => ({
                    '&': '&amp;',
                    '<': '&lt;',
                    '>': '&gt;',
                    "'": '&#39;',
                    '"': '&quot;'
                }[tag] || tag)
            );
        }
